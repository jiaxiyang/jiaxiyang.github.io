v4l2

  1. sudo v4l2-ctl --list-devices获取相机列表
  2. media-ctl -p
  3. v4l2-ctl --all -d /dev/video
  4. v4l2-ctl -d /dev/video0 --list-formats-ext 列出/dev/video0 支持的格式
  5. yavta /dev/video0 -n3 -c10 -f YUYV -s1920x1282 --skip 9 -Fvideo0_#.yuv & 从/dev/video0 拿图, 保存未.yuv
  6. v4l2-ctl --set-fmt-video=width=1920,height=1282,pixelformat=YUYV --stream-mmap --stream-count=1 -d /dev/video0 --stream-to=camera_0_1920_1282_0.yuv 获取一张图片
  7. send video to /dev/video0
  8. ffmpeg -y -t 15 -f video4linux2 -video_size 1920x1080 -input_format nv12 -timestamps abs -use_libv4l2 1 -i /dev/video0 out.mov ffmpeg link
