    samples

    1. gitlab release + jfrog artifactory（如果 docker 有问题，通过 artifacts 传递文件， 在 release 传文件和配置链接）
    stages:
  - build
  - release

variables:
  TEST_FILE0: "test0-${PACKAGE_VERSION}-${CI_COMMIT_TAG}"
  TEST_FILE1: "test1-${PACKAGE_VERSION}-${CI_COMMIT_TAG}"
  PACKAGE_VERSION: "1.2.3"
  PACKAGE_REGISTRY_URL: "http://artifacts.xxx.io/artifactory/dcv-cpp"

default:
  before_script:
    - JFROG_USER=release
    - JFROG_PASSWD=xxx
    - |
      push_to_jfrog() {
          FILE=$1;
          JFROG_PATH=http://artifacts.xxx.io/artifactory/$2;
          MD5=$(md5sum $FILE | awk '{print $1}');
          curl -u"${JFROG_USER}:${JFROG_PASSWD}" -T $FILE -H "X-Checksum-MD5:${MD5}" -L "${JFROG_PATH}";
      }

build:
  stage: build
  image: hub.xxx.io/avp/cuda11.3.1-trt8.0.3.4-ros2-desktop-ubuntu20.04:v2.0
  tags:
    - phigent_runner_gpu
  script:
    - echo $CI_COMMIT_TAG
    - mkdir bin
    - echo "${TEST_FILE0}" > bin/${TEST_FILE0}
    - echo "${TEST_FILE1}" > bin/${TEST_FILE1}
    - |
      if [ "$CI_COMMIT_TAG" != '' ]; then
        push_to_jfrog bin/${TEST_FILE0} dcv-cpp/${CI_COMMIT_TAG}/${TEST_FILE0}
        push_to_jfrog bin/${TEST_FILE1} dcv-cpp/${CI_COMMIT_TAG}/${TEST_FILE1}; fi

release:
  tags:
    - phigent_runner_gpu
  stage: release
  image: registry.gitlab.com/gitlab-org/release-cli:latest
  rules:
    - if: $CI_COMMIT_TAG
  script:
    - |
      release-cli create --name "Release $CI_COMMIT_TAG" --tag-name $CI_COMMIT_TAG \
        --assets-link "{\"name\":\"${TEST_FILE0}\",\"url\":\"${PACKAGE_REGISTRY_URL}/${CI_COMMIT_TAG}/${TEST_FILE0}\"}" \
        --assets-link "{\"name\":\"${TEST_FILE1}\",\"url\":\"${PACKAGE_REGISTRY_URL}/${CI_COMMIT_TAG}/${TEST_FILE1}\"}"
